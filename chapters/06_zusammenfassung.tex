\section{Fazit}\sectionFrame

\begin{frame}{Beantwortung der Forschungsfrage}
    \vfill
    
    \begin{center}
        \textit{Wie gut eignet sich eine modulare, containerisierte Systemarchitektur zur Integration verschiedener Echtzeit-3D-Rekonstruktionsverfahren in eine bestehende VR-Umgebung?}
    \end{center}
    
    \vfill

    \begin{itemize}
        \item[$\checkmark$] Modularität: Erfolgreiche Integration heterogener Verfahren
        \item[$\checkmark$] Echtzeitfähigkeit: Praktikable Latenzwerte in kontrollierten Szenen
        \item[$\checkmark$] Stabilität: Robuste Multi-Modell- und Multi-Client-Unterstützung
        \item[$\checkmark$] Integrationsfähigkeit: Seamless Einbindung in Va.Si.Li-Lab ohne Invasivität
    \end{itemize}

    \vfill

    \textbf{Limitierungen:}~GPU-VRAM, Szenenkomplexität, MetaQuest-API
\end{frame}
