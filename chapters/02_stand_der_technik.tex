\begin{frame}{End-to-End-Pipelines: Zentrale Limitationen}
    \vspace{0.2cm}
    
    \begin{tcolorbox}[box1, boxsep=6pt, title={\textbf{1. Monolithische Architektur}}]
        Rekonstruktionslogik fest gekoppelt, schwer austauschbar
    \end{tcolorbox}
    
    \vspace{0.4cm}
    
    \begin{tcolorbox}[box2, boxsep=6pt, title={\textbf{2. Fehlende Streaming-Infrastruktur}}]
        Batch-Processing statt Live-Datenfluss – keine Multi-User-Unterstützung
    \end{tcolorbox}
    
    \vspace{0.4cm}
    
    \begin{tcolorbox}[box3, boxsep=6pt, title={\textbf{3. Pose-Inflexibilität}}]
        Extern vs. intern: Heterogene Anforderungen können nicht parallel laufen
    \end{tcolorbox}
\end{frame}
